function pow(x, n: integer): integer;
begin
    if n = 0 then return 1;
    with res: integer; do begin
        res := pow(x, n div 2);
        res := (res * res) mod m;
        if n mod 2 = 1 then res := (res * x) mod m;
        return res;
    end
end
