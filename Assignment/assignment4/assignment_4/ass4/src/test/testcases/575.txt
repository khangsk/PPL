            isTrue, isT, isF:boolean;

            a := 10;
            b := a * 2;
            c := b div 7;
            arr[1] := c + b div 2;
            arr[2] := a div 2 + c div 3;
            arr[3] := b * a div 4 + c div 4;
            fa := (arr[1] * b div a) - (c * arr[3] - arr[2]);
            fb := a * (fa + arr[3]) * (c - arr[2]);
            fc := fa * fb / arr[2];
            isT := fa < fc;
            isF := fb >= fa;
