
            function isPrime(main:integer):boolean;
            var i: integer;
            begin
                if main < 2 then
                    return NOT TRUE;
                else 
                begin
                    if main = 2 then
                        return TRUE;
                    else
                    begin
                        for i := 2 to main - 1 do
                            if (main mod i) = 0 then
                                return FALSE;
                        return TRUE;
                    end
                end
            end
